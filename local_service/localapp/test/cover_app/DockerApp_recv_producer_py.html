def _opendb(self,filename,mode='c'): 

lockfilemode = 'a' 

lockmode = fcntl.LOCK_EX 

self.lockfd = open(filename+'.lck', lockfilemode) 

 

try: 

fcntl.flock(self.lockfd.fileno(), lockmode) 

except Exception: 

errorLogger.error('\n'+str(traceback.format_exc())+'\n') 

return (None, None) 

self.db = shelve.open(filename, flag=mode,\ 

writeback=True) 

 

def append_proc_task(self,id): 

self.proc_tasks.append(id) 

 

def _closedb(self): 

self.db.close() 

fcntl.flock(self.lockfd.fileno(),fcntl.LOCK_UN) 

self.lockfd.close() 

 

def request_new_tasks(self, devices): 

new_tasks = [] 

for device in devices: 

cur_tasks = self.client.request_tasks( 

device['user_server_no']) 

for cur_task in cur_tasks: 

exist = False 

for task in new_tasks: 

if task['id'] == cur_task['id']: 

exist = True 

break 

if not exist: 

new_tasks.append(cur_task) 

return new_tasks 

 

def request_unfin_tasks(self, devices): 

unfin_tasks = [] 

for device in devices: 

cur_tasks = self.client.request_running_tasks( 

device['user_server_no']) 

for cur_task in cur_tasks: 

exist = False 

for task in unfin_tasks: 

if task['id'] == cur_task['id']: 

exist = True 

break 

if not exist: 

cur_task['gpu_id'] = device['user_server_no'] 

unfin_tasks.append(cur_task) 

return unfin_tasks
